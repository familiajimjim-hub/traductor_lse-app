async function processText(text) { // Palabras completas const palabrasGrabadas = ["hola", "adios", "gracias"]; if (palabrasGrabadas.includes(text)) { // AHORA BUSCA EN: LSE -> videos -> archivo.mov await playVideo(`LSE/videos/${text}.mov`); } else { // Letra a letra const letras = text.replace(/\s/g, "").split(""); for (let letra of letras) { await playVideo(`LSE/videos/${letra}.mov`); } } } function playVideo(src) { return new Promise((resolve) => { const videoPlayer = document.getElementById('lse-video'); videoPlayer.src = src; videoPlayer.onended = resolve; videoPlayer.onerror = () => { console.error("No se encuentra en: ", src); resolve(); }; videoPlayer.play().catch(err => { console.error("Error:", err); resolve(); }); }); }
